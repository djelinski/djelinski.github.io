83 case TypeFunc::Control:

87 case TypeFunc::FramePtr:
88 return new MachProjNode(this,proj->_con,Matcher::c_frame_ptr_mask, Op_RegP);
89 case TypeFunc::ReturnAdr:
90 return new MachProjNode(this,proj->_con,match->_return_addr_mask,Op_RegP);
91 case TypeFunc::Parms:
92 default: {
93 uint parm_num = proj->_con - TypeFunc::Parms;
94 const Type *t = _domain->field_at(proj->_con);
95 if (t->base() == Type::Half) // 2nd half of Longs and Doubles
96 return new ConNode(Type::TOP);
97 uint ideal_reg = t->ideal_reg();
98 RegMask &rm = match->_calling_convention_mask[parm_num];
99 return new MachProjNode(this,proj->_con,rm,ideal_reg);
100 }
101 }
102 return NULL__null;
103}

1411//------------------------------Identity---------------------------------------
1412// Remove obviously duplicate safepoints
1413Node* SafePointNode::Identity(PhaseGVN* phase) {
1414
1415 // If you have back to back safepoints, remove one
1416 if (in(TypeFunc::Control)->is_SafePoint()) {
1417 Node* out_c = unique_ctrl_out();
1418 // This can be the safepoint of an outer strip mined loop if the inner loop's backedge was removed. Replacing the
1419 // outer loop's safepoint could confuse removal of the outer loop.
1420 if (out_c != NULL__null && !out_c->is_OuterStripMinedLoopEnd()) {
1421 return in(TypeFunc::Control);

1426 // eliminate a safepoint until loop opts are over.
1427 if (in(0)->is_Proj() && !phase->C->major_progress()) {

); ::breakpoint(); } } while (0);

1434 if( n0->is_Call() && n0->as_Call()->guaranteed_safepoint() ) {

1773// Redundant lock elimination
1774//
1775// There are various patterns of locking where we release and
1776// immediately reacquire a lock in a piece of code where no operations
1777// occur in between that would be observable. In those cases we can
1778// skip releasing and reacquiring the lock without violating any
1779// fairness requirements. Doing this around a loop could cause a lock
1780// to be held for a very long time so we concentrate on non-looping
1781// control flow. We also require that the operations are fully
1782// redundant meaning that we don't introduce new lock operations on
1783// some paths so to be able to eliminate it on others ala PRE. This
1784// would probably require some more extensive graph manipulation to
1785// guarantee that the memory edges were all handled correctly.
1786//
1787// Assuming p is a simple predicate which can't trap in any way and s
1788// is a synchronized method consider this code:
1789//
1790// s();
1791// if (p)
1792// s();
1793// else
1794// s();
1795// s();
1796//
1797// 1. The unlocks of the first call to s can be eliminated if the
1798// locks inside the then and else branches are eliminated.
1799//
1800// 2. The unlocks of the then and else branches can be eliminated if
1801// the lock of the final call to s is eliminated.
1802//
1803// Either of these cases subsumes the simple case of sequential control flow
1804//
1805// Addtionally we can eliminate versions without the else case:
1806//
1807// s();
1808// if (p)
1809// s();
1810// s();
1811//
1812// 3. In this case we eliminate the unlock of the first s, the lock
1813// and unlock in the then case and the lock in the final s.
1814//
1815// Note also that in all these cases the then/else pieces don't have
1816// to be trivial as long as they begin and end with synchronization
1817// operations.
1818//
1819// s();
1820// if (p)
1821// s();
1822// f();
1823// s();
1824// s();
1825//
1826// The code will work properly for this case, leaving in the unlock
1827// before the call to f and the relock after it.
1828//
1829// A potentially interesting case which isn't handled here is when the
1830// locking is partially redundant.
1831//
1832// s();
1833// if (p)
1834// s();
1835//
1836// This could be eliminated putting unlocking on the else case and
1837// eliminating the first unlock and the lock in the then side.
1838// Alternatively the unlock could be moved out of the then side so it
1839// was after the merge and the first unlock and second lock
1840// eliminated. This might require less manipulation of the memory
1841// state to get correct.
1842//
1843// Additionally we might allow work between a unlock and lock before
1844// giving up eliminating the locks. The current code disallows any
1845// conditional control flow between these operations. A formulation
1846// similar to partial redundancy elimination computing the
1847// availability of unlocking and the anticipatability of locking at a
1848// program point would allow detection of fully redundant locking with
1849// some amount of work in between. I'm not sure how often I really
1850// think that would occur though. Most of the cases I've seen
1851// indicate it's likely non-trivial work would occur in between.
1852// There may be other more complicated constructs where we could
1853// eliminate locking but I haven't seen any others appear as hot or
1854// interesting.
1855//
1856// Locking and unlocking have a canonical form in ideal that looks
1857// roughly like this:
1858//
1859// <obj>
1860// | \\------+
1861// | \ \
1862// | BoxLock \
1863// | | | \
1864// | | \ \
1865// | | FastLock
1866// | | /
1867// | | /
1868// | | |
1869//
1870// Lock
1871// |
1872// Proj #0
1873// |
1874// MembarAcquire
1875// |
1876// Proj #0
1877//
1878// MembarRelease
1879// |
1880// Proj #0
1881// |
1882// Unlock
1883// |
1884// Proj #0
1885//
1886//
1887// This code proceeds by processing Lock nodes during PhaseIterGVN
1888// and searching back through its control for the proper code
1889// patterns. Once it finds a set of lock and unlock operations to
1890// eliminate they are marked as eliminatable which causes the
1891// expansion of the Lock and Unlock macro nodes to make the operation a NOP
1892//
1893//=============================================================================
1894
1895//
1896// Utility function to skip over uninteresting control nodes. Nodes skipped are:
1897// - copy regions. (These may not have been optimized away yet.)
1898// - eliminated locking nodes
1899//
1900static Node *next_control(Node *ctrl) {
1901 if (ctrl == NULL__null)
1902 return NULL__null;
1903 while (1) {
1904 if (ctrl->is_Region()) {

1915 } else {

1917 }
1918 } else {

1920 }
1921 }
1922 return ctrl;
1923}

2038 for (int i = 1; i < (int)region->req(); i++) {
2039 Node *in_node = next_control(region->in(i));
2040 if (in_node != NULL__null) {
2041 if (find_matching_unlock(in_node, lock, lock_ops)) {
2042 // found a match so keep on checking.
2043 continue;
2044 } else if (find_lock_and_unlock_through_if(in_node, lock, lock_ops)) {
2045 continue;
2046 }
2047
2048 // If we fall through to here then it was some kind of node we
2049 // don't understand or there wasn't a matching unlock, so give
2050 // up trying to merge locks.
2051 lock_ops.trunc_to(0);
2052 return false;
2053 }
2054 }
2055 return true;
