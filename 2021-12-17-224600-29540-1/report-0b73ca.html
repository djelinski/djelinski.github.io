56unsigned int method_hash(const Method* method) {
57 unsigned int hash = method->method_holder()->class_loader_data()->identity_hash();
58 hash = (hash * 31) ^ method->klass_name()->identity_hash();
59 hash = (hash * 31) ^ method->name()->identity_hash();
60 hash = (hash * 31) ^ method->signature()->identity_hash();
61 return hash;
62}
