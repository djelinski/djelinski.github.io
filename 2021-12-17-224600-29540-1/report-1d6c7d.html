1295bool ConcreteMethodFinder::is_witness(Klass* k) {
1296 if (is_participant(k)) {
1297 return false; // do not report participant types
1298 }
1299 if (k->is_instance_klass()) {
1300 InstanceKlass* ik = InstanceKlass::cast(k);
1301 // Search class hierarchy first, skipping private implementations
1302 // as they never override any inherited methods
1303 Method* m = ik->find_instance_method(_name, _signature, Klass::PrivateLookupMode::skip);
1304 if (Dependencies::is_concrete_method(m, ik)) {
1305 return record_witness(k, m); // concrete method found
1306 } else {
1307 // Check for re-abstraction of method
1308 if (!ik->is_interface() && m != NULL__null && m->is_abstract()) {
1309 // Found a matching abstract method 'm' in the class hierarchy.
1310 // This is fine iff 'k' is an abstract class and all concrete subtypes
1311 // of 'k' override 'm' and are participates of the current search.
1312 ConcreteSubtypeFinder wf;
1313 for (uint i = 0; i < num_participants(); i++) {
1314 Klass* p = participant(i);
1315 wf.add_participant(p);
1316 }
1317 Klass* w = wf.find_witness(ik);
1318 if (w != NULL__null) {
1319 Method* wm = InstanceKlass::cast(w)->find_instance_method(_name, _signature, Klass::PrivateLookupMode::skip);
1320 if (!Dependencies::is_concrete_method(wm, w)) {
1321 // Found a concrete subtype 'w' which does not override abstract method 'm'.
1322 // Bail out because 'm' could be called with 'w' as receiver (leading to an
1323 // AbstractMethodError) and thus the method we are looking for is not unique.
1324 return record_witness(k, m);
1325 }
1326 }
1327 }
1328 // Check interface defaults also, if any exist.
1329 Array<Method*>* default_methods = ik->default_methods();
1330 if (default_methods != NULL__null) {
1331 Method* dm = ik->find_method(default_methods, _name, _signature);
1332 if (Dependencies::is_concrete_method(dm, NULL__null)) {
1333 return record_witness(k, dm); // default method found
1334 }
1335 }
1336 return false; // no concrete method found
1337 }
1338 } else {
1339 return false; // no methods to find in an array type
1340 }
